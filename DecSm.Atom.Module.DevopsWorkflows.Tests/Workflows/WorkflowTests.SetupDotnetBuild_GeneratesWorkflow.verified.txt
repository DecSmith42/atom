name: setup-dotnet

trigger:
  branches:
    include:
      - 'main'

jobs:
  
  - job: SetupDotnetTarget
    pool:
      vmImage: ubuntu-latest
    steps:
      
      - checkout: self
        fetchDepth: 0
      
      - task: UseDotNet@2
        inputs:
          version: '9.0.x'

      - script: dotnet run --project AtomTest/AtomTest.csproj SetupDotnetTarget --skip --headless
        name: SetupDotnetTarget
